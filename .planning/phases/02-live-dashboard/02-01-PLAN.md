Add a <script> block to hooch-snapshot.html immediately before the closing </body> tag. The entire script is an IIFE wrapping all logic. Use 'use strict'. All DOM insertion must use textContent — never innerHTML.

Insert this exact script:

  <script>
  (function() {
    'use strict';

    var STATIONS = ['02334430','02335000','02335450','02335815','02335880','02336000','02337170','02338000','02338500'];
    var PARAMS = ['00065','00060','00010','63680'];
    var API_URL = 'https://waterservices.usgs.gov/nwis/iv/?sites=' + STATIONS.join(',') + '&parameterCd=' + PARAMS.join(',') + '&format=json';
    var STALE_MS = 2 * 60 * 60 * 1000;

    function getReading(map, stationId, paramCode) {
      var key = 'USGS:' + stationId + ':' + paramCode + ':00000';
      var ts = map[key];
      if (!ts) return null;
      var reading = ts.values[0].value[0];
      if (!reading || reading.value === '-999999') return null;
      if ((Date.now() - new Date(reading.dateTime).getTime()) > STALE_MS) return null;
      return reading.value;
    }

    function populateCard(stationId, map) {
      var card = document.querySelector('.card[data-station="' + stationId + '"]');
      if (!card) return;

      var idSpan = card.querySelector('.station-id');
      if (idSpan) idSpan.textContent = stationId;

      card.querySelectorAll('.stat-row[data-param]').forEach(function(row) {
        var paramCode = row.dataset.param;
        var rawValue = getReading(map, stationId, paramCode);
        if (rawValue === null) { row.remove(); return; }
        var valueSpan = row.querySelector('.value');
        var display = rawValue;
        var unit = '';
        if (paramCode === '00065') unit = ' ft';
        else if (paramCode === '00060') unit = ' cfs';
        else if (paramCode === '00010') {
          display = ((parseFloat(rawValue) * 9 / 5) + 32).toFixed(1);
          unit = ' °F';
        }
        else if (paramCode === '63680') unit = ' FNU';
        valueSpan.textContent = display + unit;
        valueSpan.style.color = 'var(--color-text)';
      });
    }

    function showError(msg) {
      var banner = document.createElement('div');
      banner.setAttribute('role', 'alert');
      banner.style.cssText = 'background:var(--color-destructive);color:#fff;padding:var(--space-sm) var(--space-xl);font-size:var(--text-body);';
      banner.textContent = msg;
      document.body.insertBefore(banner, document.body.firstChild);
    }

    function loadData() {
      fetch(API_URL)
        .then(function(res) {
          if (!res.ok) throw new Error('HTTP ' + res.status);
          return res.json();
        })
        .then(function(json) {
          var map = {};
          json.value.timeSeries.forEach(function(ts) { map[ts.name] = ts; });
          STATIONS.forEach(function(id) { populateCard(id, map); });
        })
        .catch(function() {
          showError('Unable to load river conditions. Check your internet connection.');
        });
    }

    document.querySelectorAll('.card[data-station]').forEach(function(card) {
      var stationId = card.dataset.station;
      card.style.cursor = 'pointer';
      card.addEventListener('click', function() {
        window.open('https://waterdata.usgs.gov/monitoring-location/' + stationId + '/', '_blank', 'noopener,noreferrer');
      });
    });

    loadData();
  })();
  </script>

Notes on the implementation above:
- Temperature unit uses the Unicode escape °F to avoid encoding issues in the file. This produces "°F" in the browser.
- async/await is avoided in favor of .then()/.catch() for maximum compatibility with old browsers opened via file:// — both are correct; the .then() chain matches the IIFE style and avoids any potential transpilation confusion.
- The IIFE runs synchronously on script evaluation (placed before </body>, so DOM is fully parsed at that point). No DOMContentLoaded listener is needed.
- Do not use innerHTML anywhere in this script. All dynamic content uses textContent or createElement + style assignments.
  </action>

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → USGS waterservices.usgs.gov | Untrusted JSON response enters the DOM via JS. The response is public and CORS-open. A compromised or MITM'd response could contain crafted strings. |
| Browser → spatialdata.usace.army.mil | The USACE page is loaded in an iframe. The browser enforces X-Frame-Options; no content escapes into the parent page. |
| Browser → waterdata.usgs.gov | Users are navigated to this URL via window.open. The URL is constructed from hardcoded constants, not external input. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
